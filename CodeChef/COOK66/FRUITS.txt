#include<stdio.h>
	int main()
	{
	int t;
	scanf("%d",&t);
	while(t-->0)
	{
	int a,b,k;
	scanf("%d%d%d",&a,&b,&k);
	if(a==b)
	{
	if(k%2==0)
	{a=0;
	b=0;}
	else
	{
	a=1;b=0;
	}
	}
	else
	while(k-->0)
	{
	if(a>b)
	a=a-1;
	else
	b=b-1;
	}
	int p=a-b;
	if(p<0)
	p=-p;
	printf("%d\n",p);
	}
	 
	return 0;
	 
	 
	}
