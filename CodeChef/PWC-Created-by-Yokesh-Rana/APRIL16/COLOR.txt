#include<stdio.h>
	int main()
	{
	int t;
	scanf("%d",&t);
	while(t-->0)
	{
	int n;
	scanf("%d",&n);
	char st[100000];
	scanf("%s",st);
	int r=0,g=0,b=0;
	int i=0;
	while(st[i]!=0)
	{
	if(st[i]=='R')
	r++;
	if(st[i]=='G')
	g++;
	if(st[i]=='B')
	b++;
	i++;
	}
	if(r>=g&&r>=b)
	{
	printf("%d\n",g+b);
	continue;
	}
	if(g>=r&&g>=b)
	{
	printf("%d\n",r+b);
	continue;
	}
	if(b>=g&&b>=r)
	{
	printf("%d\n",g+r);
	continue;
	}
	
	}
	return 0;
	}
