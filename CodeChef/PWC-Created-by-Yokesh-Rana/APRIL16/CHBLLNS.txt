#include<stdio.h>
	int main()
	{
	long int t;
	scanf("%ld",&t);
	while(t-->0)
	{
	long int count=0,r,g,b;
	scanf("%ld%ld%ld",&r,&g,&b);
	int R=0,G=0,B=0;
	int k;
	scanf("%ld",&k);
	while(1)
	{
	if(R==k||G==k||B==k)
	break;
	if(r>0)
	{
	r--;
	R++;
	count++;
	}
	if(R==k||G==k||B==k)
	break;
	if(g>0)
	{
	g--;
	G++;
	count++;
	}
	if(R==k||G==k||B==k)
	break;
	if(b>0)
	{
	b--;
	B++;
	count++;
	}
	
	}
	printf("%ld\n",count);
	}
	return 0;
	}
