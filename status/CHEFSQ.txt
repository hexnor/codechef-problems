#include<stdio.h>
	int main()
	{
	int t;
	scanf("%d",&t);
	while(t-->0)
	{
	int n;
	scanf("%d",&n);
	int a[n],i,j;
	for(i=0;i<n;i++)
	scanf("%d",a+i);
	int p;
	scanf("%d",&p);
	int s[p];
	for(i=0;i<p;i++)
	scanf("%d",s+i);
	
	int count=0;
	for(j=0;j<p;j++)
	for(i=0;i<n;i++)
	{
	if(s[j]==a[i])
	{
	count++;i=n;}
	}
	if(count>=p)
	printf("Yes\n");
	else
	printf("No\n");
	
	}
	
	return 0;
	
	
	
	
	
	}
