#include<stdio.h>
	#include<string.h>
	main()
	{
	int t;
	scanf("%d",&t);
	 
	while(t-->0)
	{
	char a[50],b[50],i;
	scanf("%s%s",&a,&b);
	 
	int flag=1;
	for(i=0;i<strlen(a);i++)
	if(a[i]==b[i])
	continue;
	else
	{
	if(a[i]=='?'||b[i]=='?')
	continue;
	flag=0;
	}
	if(flag==0)
	printf("No\n");
	else
	printf("Yes\n");
	 
	}
	 
	}
