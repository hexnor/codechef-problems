#include<stdio.h>
	#include<math.h>
	int main()
	{
	int t,bc[21],i,j,n,m;
	scanf("%d",&t);
	while(t-->0)
	{
	int flag=0;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
	scanf("%d",&bc[i]);
	for(i=0;i<pow(2,n);i++)
	{ int sum=0;
	for(j=0;j<n;j++)
	{
	int k=i>>j;
	if(k&1)
	sum+=bc[j];
	if(sum==m)
	flag=1;
	if(flag==1)
	break;
	}
	}
	if(flag==1)
	printf("YES\n");
	else
	printf("NO\n");
	}
	return 0;
	}
