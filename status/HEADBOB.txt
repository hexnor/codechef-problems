#include<stdio.h>
	#include<string.h>
	int main()
	{
	int t;
	scanf("%d",&t);
	while(t-->0)
	{
	int n;
	scanf("%d",&n);
	int in=0,noin=0,no=0;
	char a[n];
	int i;
	scanf("%s",a);
	for(i=0;i<strlen(a);i++)
	{
	if(a[i]=='I')
	in=1;
	if(a[i]=='Y')
	noin=1;
	if(a[i]=='N')
	no=1;
	}
	if(in==0&&noin==0&&no==1)
	printf("NOT SURE\n");
	if(in==1)
	printf("INDIAN\n");
	if(noin==1&&in==0)
	printf("NOT INDIAN\n");
	 
	 
	 
	 
	}
	 
	}
