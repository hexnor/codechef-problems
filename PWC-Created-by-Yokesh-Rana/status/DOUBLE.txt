#include<stdio.h>
	 
	int main()
	 
	{
	 
	register int i;
	 
	int n;
	 
	scanf("%d",&n);
	 
	int a[n];
	 
	for(i=0;i<n;i++)
	 
	scanf("%d",a+i);
	 
	for(i=0;i<n;i++)
	 
	{
	 
	if(a[i]&0==0)
	 
	printf("%d\n",a[i]-1);
	 
	else printf("%d\n",a[i]);
	 
	}
	 
	return 0;
	 
	
	}
