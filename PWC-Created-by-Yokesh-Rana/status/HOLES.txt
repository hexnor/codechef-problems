#include<stdio.h>
	 
	int main()
	 
	{
	 
	register int i;
	 
	int n;
	 
	scanf("%d",&n);
	 
	for(i=0;i<n;i++)
	 
	{
	 
	char a[100];
	 
	scanf("%s",a);
	 
	int count=0,i=0;
	 
	
	 
	while(a[i]!=0)
	 
	{
	 
	if(a[i]=='A'||a[i]=='D'||a[i]=='O'||a[i]=='R'||a[i]=='P'||a[i]=='Q')
	 
	count++;
	 
	else
	 
	if(a[i]=='B')
	 
	count=count+2;
	 
	i++;
	 
	}
	 
	printf("%d\n",count);
	 
	}
	 
	
	return 0;
	 
	
	 
	}
