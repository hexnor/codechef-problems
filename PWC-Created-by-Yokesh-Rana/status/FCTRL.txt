#include<stdio.h>
	int main()
	{
	long int n,z,i;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
	long int c=0;
	scanf("%d",&z);
	while(z>0)
	{
	c+=z/5;
	z=z/5;
	}
	printf("%d\n",c);
	 
	 
	}
	return 0;
	}
